001    package org.codemonkey.swiftsocketclient;
002    
003    import org.codemonkey.util.DatagramEncoder;
004    
005    /**
006     * Provides a default {@link DatagramEncoder} implementation that returns an empty string. The {@link #encode()} method implemented can be
007     * used (by not overriding it in sub classes) when only the message id is important (for example with Ping Pong messages, which have no
008     * content).
009     * <p>
010     * With this base class a message may encode itself to a <code>String</code>. This interface is used to translate messages into so called <a
011     * href="http://gpwiki.org/index.php/Binary_Packet">Binary Packets</a>, or datagrams.
012     * <p>
013     * A response string needs to built up in the following format:<br />
014     * message code (3), encoded string
015     * <p>
016     * The following example could of a login response: <code>005OK</code><br />
017     * The following example could of a pong notification: <code>999</code><br />
018     * 
019     * @author Benny Bottema
020     * @see SwiftSocketClient#addServerResponse(ServerMessageToClient)
021     * @see SwiftSocketClient#getNextServerResponse()
022     * @see DatagramEncoder#encode()
023     * @since 1.0
024     */
025    public abstract class ClientMessageToServer implements DatagramEncoder {
026    
027            /**
028             * Default implementation which returns an empty string. This is useful for messages that have no parameters, just a message code.
029             * 
030             * @see DatagramEncoder#encode()
031             */
032            @Override
033            public String encode() {
034                    return "";
035            }
036    }
