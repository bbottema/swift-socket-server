001    package org.codemonkey.swiftsocketclient;
002    
003    import java.io.IOException;
004    import java.net.DatagramSocket;
005    import java.net.Socket;
006    import java.net.SocketException;
007    
008    /**
009     * Defines the type of client, {@link #UDP} or {@link #TCP}. Determines which {@link ClientEndpoint} is being used. The client behaves
010     * exactly the same in both modes, except the message transport types are being switched (without breaking API). In UDP mode, TCP mode is
011     * being emulated (see {@link ClientEndpoint} for more info on this).
012     * 
013     * @author Benny Bottema
014     * @since 1.0
015     */
016    public enum ClientType {
017            /**
018             * Creates a {@link ClientEndpointUDP} with an embedded {@link DatagramSocket}.
019             * 
020             * @see ClientEndpointUDP
021             * @see DatagramSocket#DatagramSocket()
022             */
023            UDP {
024                    /**
025                     * @see ClientType#UDP
026                     */
027                    @Override
028                    ClientEndpoint createClientEndpoint(final String host, final int port)
029                                    throws SocketException {
030                            return new ClientEndpointUDP(new DatagramSocket(), host, port);
031                    }
032            },
033            /**
034             * Creates a {@link ClientEndpointTCP} with an embedded {@link Socket}.
035             * 
036             * @see ClientEndpointTCP
037             * @see Socket#Socket(String, int)
038             */
039            TCP {
040    
041                    /**
042                     * @see ClientType#TCP
043                     */
044                    @Override
045                    ClientEndpoint createClientEndpoint(final String host, final int port)
046                                    throws IOException {
047                            return new ClientEndpointTCP(new Socket(host, port));
048                    }
049            };
050    
051            /**
052             * Creates a transport type specific {@link ClientEndpoint}, either TCP or UDP.
053             */
054            abstract ClientEndpoint createClientEndpoint(String host, int port)
055                            throws SocketException, IOException;
056    }
