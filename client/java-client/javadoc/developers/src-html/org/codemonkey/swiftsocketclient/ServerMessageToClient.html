001    package org.codemonkey.swiftsocketclient;
002    
003    import org.codemonkey.util.Executable;
004    
005    /**
006     * Defines a base for decodable and executable message types.
007     * <p>
008     * With this base class, a message can decode a <code>String</code> to fill itself. This interface is used to translate so called <a
009     * href="http://gpwiki.org/index.php/Binary_Packet">Binary Packets</a>, or datagrams, into {@link Executable} messages. This means an
010     * instance of {@link ClientMessageToServer} is very much a command-like object.
011     * <p>
012     * A response string needs to built up in the following format:<br />
013     * message id (3), encoded string
014     * <p>
015     * The following example could of a login response: <code>0035admin7letmein</code><br />
016     * The following example could of a pong notification: <code>999</code><br />
017     * <p>
018     * It is up to the implementing class to decode this information and utilize it as an executable object.
019     * <p>
020     * 
021     * @author Benny Bottema
022     * @param <Controller> The type to which the executable can defer to (also see {@link Executable#execute(Object)}).
023     * @see Executable The interface used by the server to execute the client message after it has been decoded.
024     * @see SwiftSocketClient#addServerResponse(ServerMessageToClient)
025     * @see SwiftSocketClient#getNextServerResponse()
026     * @since 1.0
027     */
028    public interface ServerMessageToClient<Controller> extends Executable<Controller, UnknownMessageException> {
029            /**
030             * Populates itself by decoding the given message content. How this is done is entirely up to the sub class implementation, as long as
031             * it is decoded the same way it was encoded.
032             * 
033             * @param messageContent The string containing a message id and the message content to be decoded.
034             */
035            public abstract void decode(final String messageContent);
036    }
