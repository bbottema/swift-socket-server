001    package org.codemonkey.swiftsocketclient;
002    
003    /**
004     * Wrapper class of which instances substitute bad client-to-server messages. Created when decoding a message datagram failed.
005     * <p>
006     * The error is communicated when the message is being executed by the server, but does not throw an exception.
007     * 
008     * @see ServerHandler#createDecodedExecutableResponse(String)
009     * @author Benny Bottema
010     * @since 1.0
011     */
012    class ServerMessageToClientInvalid implements ServerMessageToClient<Object> {
013    
014            private String message = "";
015    
016            private final Exception failureCause;
017    
018            public ServerMessageToClientInvalid(final ServerContext serverContext, final Exception failureCause) {
019                    this.failureCause = failureCause;
020            }
021    
022            /**
023             * {@inheritDoc}
024             */
025            @Override
026            public void decode(final String message) {
027                    this.message = message;
028            }
029    
030            /**
031             * Throws a {@link UnknownMessageException} with the current <code>ServerMessageToClientInvalid</code> instance.
032             */
033            @Override
034            public void execute(final Object server)
035                            throws UnknownMessageException {
036                    throw new UnknownMessageException(message, failureCause);
037            }
038    }
