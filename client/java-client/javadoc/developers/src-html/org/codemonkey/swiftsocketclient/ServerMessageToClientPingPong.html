001    package org.codemonkey.swiftsocketclient;
002    
003    /**
004     * Represents a server Ping message and when executed responds with a Pong message via {@link ServerHandler#pong()}.
005     * 
006     * @author Benny Bottema
007     * @since 1.0
008     */
009    final class ServerMessageToClientPingPong implements ServerMessageToClient<ServerHandler> {
010            
011            /**
012             * Validates that there is indeed no content in this message, since only the message id is relevant.
013             */
014            @Override
015            public void decode(final String messageContent) {
016                    if (messageContent.length() > 0) {
017                            throw new RuntimeException("invalid pong message, reason: message too long");
018                    }
019            }
020    
021            /**
022             * Simply delegates the ping event to the {@link ServerHandler}.
023             * 
024             * @param handler The {@link ServerHandler} instance to which the pong notification can be delegated to.
025             * @see ServerHandler#pong()
026             * @see ServerMessageToClient#execute(Object)
027             */
028            @Override
029            public void execute(final ServerHandler handler) {
030                    handler.pong();
031            }
032    }
