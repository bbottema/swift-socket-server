001    package org.codemonkey.util;
002    
003    /**
004     * Interface for marking objects encodable as datagram. An encoded strings is a string that contains a number that denotes the length of a
005     * value followed by the value itself.
006     * 
007     * @author Benny Bottema
008     * @since 1.0
009     */
010    public interface DatagramEncoder {
011    
012            /**
013             * Creates a response string (binary packet) of the current server response. It does not know the response code, which is assigned
014             * externally and prepended to the response string created by this method.
015             * 
016             * @return The response <code>String</code> or binary packet, without the response code.
017             */
018            String encode();
019    }
