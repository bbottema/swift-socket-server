001    package org.codemonkey.swiftsocketclient;
002    
003    import org.apache.log4j.Logger;
004    
005    /**
006     * Helper class mainly to hide some boilerplate code concerning {@link Thread#sleep(long)}.
007     * 
008     * @author Benny Bottema
009     * @since 1.0
010     */
011    public final class ServerUtil {
012            private static final Logger LOGGER = Logger.getRootLogger();
013    
014            private ServerUtil() {
015            }
016    
017            /**
018             * Makes the current Thread sleep any number of milliseconds.
019             * 
020             * @param milliseconds The number of milliseconds to make the thread sleep for.
021             * @see Thread#sleep(long)
022             */
023            public static void sleep(final int milliseconds) {
024                    try {
025                            Thread.sleep(milliseconds);
026                    } catch (final InterruptedException e) {
027                            assert false;
028                            LOGGER.error(e.getMessage(), e);
029                    }
030            }
031    
032            /**
033             * Performs a {@link Thread#sleep(long)} of 10 milliseconds. Used to avoid using 100% of cpu.
034             */
035            public static void defaultSleep() {
036                    sleep(10);
037            }
038    }
