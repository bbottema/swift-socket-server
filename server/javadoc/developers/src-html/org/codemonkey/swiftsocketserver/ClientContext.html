001    package org.codemonkey.swiftsocketserver;
002    
003    import java.net.InetAddress;
004    import java.util.Date;
005    import java.util.HashMap;
006    import java.util.Map;
007    
008    /**
009     * Contains all session data concerning a given client. Also contains a session data container for external use only. External users can
010     * utilize this container to store client related (session) data for the duration of the connection.
011     * <p>
012     * Only the client's id and the session data container is exposed publicly. The client id is determined during construction so that even
013     * when the connection is lost, external users can still access the client's id without connection exceptions.
014     * <p>
015     * Since {@link #clientInetAddress} does not have to be unique (multiple clients connecting from the one host), instances of this classes
016     * also act as unique identifiers for client sockets so that the server can track multiple clients from the same host without exposing the
017     * Socket instance itself. Doing so allows executable message objects to access a Client's session data (or send responses to a specific
018     * client) without needing the Socket or InetAddress to identify the client.
019     * <p>
020     * This context keeps information about the client which may remain accessible even when the client has disconnected.
021     * 
022     * @author Benny Bottema
023     * @since 1.0
024     */
025    public class ClientContext {
026    
027            /**
028             * Client to which this context is dedicated to. Used to read from and write to.
029             */
030            private final ClientEndpoint clientEndpoint;
031    
032            /**
033             * Client id to which context is dedicated to. Acquired from {@link #clientInetAddress}. Kept separate for when the socket has been
034             * closed already and the user still wants to know the original client id.
035             */
036            private final InetAddress clientInetAddress;
037    
038            /**
039             * Flag that indicated whether the client has sent a 'Bye Bye' notification to gracefully close the connection, or when the connection
040             * was lost somehow.
041             */
042            private boolean clientSaidByeBye;
043    
044            /**
045             * Time stamp of the last ping message to the client.
046             * 
047             * @see #pongReceived
048             * @see #isPongReceived()
049             */
050            private long pingtime;
051    
052            /**
053             * Flag that indicates whether the client has sent a 'pong' notification.
054             * 
055             * @see #pingtime
056             * @see #isPongReceived()
057             */
058            private boolean pongReceived;
059    
060            /**
061             * A session scoped data container for external use only. Users can access this container to store temporary data concerning the current
062             * client, until the connection is lost.
063             */
064            private final Map<Object, Object> sessionData = new HashMap<Object, Object>();
065    
066            /**
067             * Stores the given client endpoint and saves its {@link InetAddress} for later use (in case the endpoint is being closed and we lose
068             * the ability the query for the {@link InetAddress}).
069             * <p />
070             * Furthermore initializes a timestamp to start measuring ping/pong timeouts.
071             * 
072             * @param clientEndpoint The client with which we have established a communication channel.
073             */
074            public ClientContext(final ClientEndpoint clientEndpoint) {
075                    this.clientEndpoint = clientEndpoint;
076                    clientInetAddress = clientEndpoint.getInetAddress();
077                    pingtime = new Date().getTime();
078                    pongReceived = false;
079            }
080    
081            /**
082             * {@inheritDoc}
083             */
084            @Override
085            public String toString() {
086                    return String.valueOf(clientEndpoint.getInetAddress());
087            }
088    
089            /**
090             * @return {@link #clientEndpoint}.
091             */
092            final ClientEndpoint getClientEndpoint() {
093                    return clientEndpoint;
094            }
095    
096            /**
097             * @return {@link #clientInetAddress}
098             */
099            public InetAddress getClientInetAddress() {
100                    return clientInetAddress;
101            }
102    
103            /**
104             * @return {@link #sessionData}
105             */
106            public Map<Object, Object> getSessionData() {
107                    return sessionData;
108            }
109    
110            /**
111             * @return {@link #pingtime}
112             */
113            final long getPingtime() {
114                    return pingtime;
115            }
116    
117            final void setPingtime(final long pingtime) {
118                    this.pingtime = pingtime;
119            }
120    
121            /**
122             * @return {@link #pongReceived}
123             */
124            final boolean isPongReceived() {
125                    return pongReceived;
126            }
127    
128            final void setPongReceived(final boolean pongReceived) {
129                    this.pongReceived = pongReceived;
130            }
131    
132            /**
133             * @return {@link #clientSaidByeBye}
134             */
135            final boolean isClientSaidByeBye() {
136                    return clientSaidByeBye;
137            }
138    
139            /**
140             * Indicates whether the client has gone away or that the connection has been closed.
141             * 
142             * @return !{@link #clientSaidByeBye} && ! {@link ClientEndpoint#isClosed()}
143             */
144            public final boolean isActive() {
145                    return !isClientSaidByeBye() && !clientEndpoint.isClosed();
146            }
147    
148            /**
149             * Sets {@link #clientSaidByeBye}. In case of 'true', the socket should be closed as well in one atomic operation, to avoid reading out
150             * a closed socket while setting Bye Bye == true.
151             * 
152             * @param clientSaidByeBye Whether the connection has become obsolete, either because the client left gracefully or by connection error.
153             */
154            final void setClientSaidByeBye(final boolean clientSaidByeBye) {
155                    this.clientSaidByeBye = clientSaidByeBye;
156            }
157    }
