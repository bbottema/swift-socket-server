001    package org.codemonkey.swiftsocketserver;
002    
003    import java.io.BufferedReader;
004    import java.io.IOException;
005    import java.lang.reflect.InvocationTargetException;
006    import java.net.Socket;
007    import java.net.SocketException;
008    import java.util.Date;
009    
010    import org.apache.commons.lang.StringUtils;
011    import org.apache.log4j.Logger;
012    
013    /**
014     * Worker thread for the server that is dedicated to a single client.
015     * <p>
016     * Sets up a child thread that listens for client input, while pinging the client to test connection. Since Java cannot detect if a client
017     * closes the TCP connection, we manually test the connection by pinging the client. If the client does not respond with
018     * {@link #pingPongTimeoutMs}, the server assumes the connection went dead.
019     * 
020     * @author Benny Bottema
021     * @see InputHandler
022     * @see SwiftSocketServer
023     * @see #checkPing()
024     * @see #checkPong()
025     * @since 1.0
026     */
027    public class ClientHandler implements Runnable {
028    
029            /**
030             * The length of the message id of a message string (binary packet).
031             */
032            protected static final int MESSAGEID_LENGTH = 3;
033    
034            /**
035             * Interval in milliseconds for ping messages to the client.
036             * 
037             * @see #checkPing()
038             * @see #checkPong()
039             */
040            private int pingPongIntervalMs;
041    
042            /**
043             * Timeout in milliseconds for a pong message to the client.
044             * 
045             * @see #checkPing()
046             * @see #checkPong()
047             */
048            private int pingPongTimeoutMs = 5000;
049    
050            /**
051             * Basic logger to log events such as client timeouts and other client related events.
052             */
053            private static final Logger LOGGER = Logger.getLogger(ClientHandler.class);
054    
055            /**
056             * The {@link SwiftSocketServer} instance used to check server status and acquire {@link ClientMessageToServer} messages from.
057             */
058            private final SwiftSocketServer server;
059    
060            /**
061             * The {@link ClientContext} containing all session data concerning the client. Also contains a session data container for external use
062             * only. External users can utilize this container to store client related data for the duration of the connection.
063             */
064            private final ClientContext clientContext;
065    
066            private final boolean pingPongMode;
067    
068            /**
069             * Constructor, which accepts a {@link SwiftSocketServer} instance and a client {@link ClientContext}. Also resets
070             * {@link ClientContext#getPingtime()} and {@link ClientContext#isPongReceived()} flags.
071             * 
072             * @param server Server used to check on the server status (ie. server is stopping).
073             * @param pingPongMode Indicates whether to perform PingPong polling to the client to detect drop outs.
074             * @param clientContext The context containing the client socket to which this worker thread is dedicated to.
075             * @param pingPongIntervalMs The Ping Pong interval used if {@link #pingPongMode} is turned on.
076             * @param pingPongTimeoutMs The Ping Pong timeout used if {@link #pingPongMode} is turned on.
077             */
078            protected ClientHandler(final SwiftSocketServer server, final ClientContext clientContext, final boolean pingPongMode,
079                            int pingPongIntervalMs, int pingPongTimeoutMs) {
080                    this.server = server;
081                    this.clientContext = clientContext;
082                    this.pingPongMode = pingPongMode;
083                    this.pingPongIntervalMs = pingPongIntervalMs;
084                    this.pingPongTimeoutMs = pingPongTimeoutMs;
085            }
086    
087            /**
088             * Thread method that runs ping messages to the client and tests for timeouts. Also sets up client thread that waits for input; a new
089             * thread is used since this action is blocking and the thread 'sleeps' until input is available, which would deny us the opportunity to
090             * do ping messages.<br />
091             * <br />
092             * Keeps pinging the client while the server is running.
093             * 
094             * @see java.lang.Runnable#run()
095             * @see #checkPing()
096             * @see #checkPong()
097             * @see SwiftSocketServer#isRunning()
098             */
099            @Override
100            public final void run() {
101                    final Thread inputThread = new Thread(new InputHandler());
102                    inputThread.start();
103    
104                    if (pingPongMode) {
105                            ping();
106                    }
107                    while (server.isRunning()) {
108                            if (!clientContext.isClientSaidByeBye()) {
109                                    if (pingPongMode) {
110                                            if (checkPong()) {
111                                                    checkPing();
112                                            } else {
113                                                    LOGGER.debug(String.format("dropping client %s, reason: timeout", clientContext));
114                                                    server.disposeOfClient(clientContext);
115                                                    break;
116                                            }
117                                    } else {
118                                            // PingPong mode is turned off, we won't be dropping clients due to time-outs
119                                    }
120                                    ServerUtil.defaultSleep(); // avoid hogging cpu
121                            } else {
122                                    LOGGER.debug(String.format("dropping client %s, reason: client went away", clientContext.getClientInetAddress()));
123                                    break;
124                            }
125                    }
126            }
127    
128            /**
129             * Sends a ping message to the client, which on turn should respond with a 'pong' notification. A ping is sent only if the following
130             * requirements are met:
131             * <ol>
132             * <li>A pong response has been received; we don't want to send out multiple ping messages</li>
133             * <li>The last ping message time stamp exceeds {@link #PING_INTERVAL}</li>
134             * </ol>
135             * 
136             * @see #ping()
137             * @see #pong()
138             * @see ClientContext#getPingtime()
139             * @see ClientContext#isPongReceived()
140             * @see #PINGPONG_TIMEOUT
141             * @see #PING_INTERVAL
142             * @see #checkPong()
143             */
144            private void checkPing() {
145                    final long pongDelay = new Date().getTime() - clientContext.getPingtime();
146                    if (clientContext.isPongReceived() && pongDelay > pingPongIntervalMs) {
147                            ping();
148                    }
149            }
150    
151            /**
152             * Evaluates whether the client has sent the last pong message, or else if the client has still time to respond with a pong
153             * notification. In other words: "returns that the client hasn't timed out yet".
154             * 
155             * @return Whether pong hasn't timed out yet.
156             * @see #ping()
157             * @see #pong()
158             * @see ClientContext#getPingtime()
159             * @see ClientContext#isPongReceived()
160             * @see #PINGPONG_TIMEOUT
161             * @see #PING_INTERVAL
162             * @see #checkPing()
163             */
164            private boolean checkPong() {
165                    return clientContext.isPongReceived() || new Date().getTime() - clientContext.getPingtime() < pingPongTimeoutMs;
166            }
167    
168            /**
169             * Performs a ping message to the client, by using a {@link ServerMessageToClientPingPong}. Resets the response flags so we can keep
170             * track of the time between the ping message and the pong notification.
171             * 
172             * @see ClientContext#getPingtime()
173             * @see ClientContext#isPongReceived()
174             */
175            private void ping() {
176                    sendMessage(new ServerMessageToClientPingPong(null));
177                    // reset pong response flags
178                    clientContext.setPingtime(new Date().getTime());
179                    clientContext.setPongReceived(false);
180    
181            }
182    
183            /**
184             * Invoked by {@link ClientMessageToServerPingPong#execute(ClientHandler)}, when the client is responding with a <em>pong</em>
185             * notification. Sets {@link ClientContext#setPongReceived(boolean)} to <code>true</code>, which is used to determine <em>ping</em>
186             * timeout with.
187             * 
188             * @see ClientContext#setPongReceived(boolean)
189             * @see #checkPong()
190             * @see ClientMessageToServerPingPong
191             */
192            public final void pong() {
193                    clientContext.setPongReceived(true);
194            }
195    
196            /**
197             * Called by {@link ClientMessageToServerByeBye#execute(ClientHandler)}. Set the 'Bye Bye' lag on the {@link #clientContext} and calls
198             * {@link ClientEndpoint#close()}.
199             */
200            protected void clientSaysByeBye() {
201                    server.disposeOfClient(clientContext);
202                    try {
203                            clientContext.getClientEndpoint().close();
204                    } catch (final IOException e) {
205                            throw new RuntimeException("error closing socket (already closed?)", e);
206                    }
207            }
208    
209            /**
210             * Logs and sends a response message to the current client using {@link #clientContext}, delegating the actual sending to
211             * {@link #sendMessage(ClientContext, int, ServerMessageToClient)}, unless the client has said 'Bye Bye' in which case the message to be
212             * sent is ignored altogether.
213             * 
214             * @param message The server message to send to the client.
215             * @see ClientContext#isClientSaidByeBye()
216             * @see #sendMessage(ClientContext, int, ServerMessageToClient)
217             */
218            public void sendMessage(final ServerMessageToClient message) {
219                    final Logger messageLogger = determineLogger(message);
220                    messageLogger.debug(String.format("sending message to %s: %s", clientContext.getClientInetAddress(), message.getClass()
221                                    .getSimpleName()));
222                    if (!clientContext.isClientSaidByeBye()) {
223                            final int messageId = server.getServerMessageId(message.getClass());
224                            sendMessage(clientContext, messageId, message);
225                    } else {
226                            // log message, giving class logger priority
227                            Logger availableLogger = LOGGER;
228                            if (!LOGGER.isDebugEnabled() && messageLogger.isDebugEnabled()) {
229                                    availableLogger = messageLogger;
230                            }
231                            availableLogger.debug(String.format("dropping message %s, reason: client already said byebye", message));
232                    }
233            }
234    
235            /**
236             * Helper method used to send a message to the client.
237             * 
238             * @param clientContext The client to send the message to.
239             * @param messageId The message id, known to both the server and the client.
240             * @param message The {@link ClientMessageToServer} used to physically encode the message as <code>String</code>.
241             */
242            private void sendMessage(final ClientContext clientContext, final int messageId, final ServerMessageToClient message) {
243                    final String paddedMessageId = StringUtils.leftPad(String.valueOf(messageId), MESSAGEID_LENGTH, '0');
244                    final String datagram = String.format("%s%s\n\0", paddedMessageId, message.encode());
245                    try {
246                            clientContext.getClientEndpoint().send(datagram);
247                    } catch (final IOException e) {
248                            throw new RuntimeException(String.format("unable to send message to client %s", clientContext.getClientInetAddress()), e);
249                    }
250            }
251    
252            /**
253             * Returns the class logger {@link #LOGGER} in case of user defined message types, or a message type specific logger if it is a native
254             * message type. This is so that we're able to differentiate log levels between Ping Pong messages (to mute them effectively) and other
255             * messages we'd still like to see in out logging.
256             * 
257             * @param object The object doing the logging.
258             * @return The correct logger to avoid using log4j on end user classes (who may not support log4j and get due warnings) and still be
259             *         able to mute specific native message types, such as {@link ClientMessageToServerPingPong}.
260             */
261            private Logger determineLogger(final Object object) {
262                    if (object.getClass().getPackage() == ClientHandler.class.getPackage()) {
263                            return Logger.getLogger(object.getClass());
264                    } else {
265                            return LOGGER;
266                    }
267            }
268    
269            /**
270             * @return The {@link #clientContext} containing session data for external use.
271             */
272            public ClientContext getClientContext() {
273                    return clientContext;
274            }

276            /**
277             * Listens to client and converts input to objects of type {@link ClientMessageToServer}. A thread is being used for this, since reading
278             * input is blocking, meaning the thread 'sleeps' while waiting for input. These messages are then added to the client messages queue
279             * for execution by external user.<br />
280             * <strong>Note:</strong> some messages are filtered by the server as they are of server level, such as pong messages.<br />
281             * <br />
282             * Reuses fields from {@link ClientHandler}, such as {@link ClientHandler#clientContext} and {@link ClientHandler#server}. Keeps
283             * listening for input while the server is still running and the client hasn't timed out.
284             * 
285             * @author Benny Bottema
286             * @since 1.0
287             */
288            private class InputHandler implements Runnable {
289    
290                    private static final String ERROR_SOCKET_CLOSED = "socket closed";
291                    private static final String ERROR_RECV_FAILED = "Software caused connection abort: recv failed";
292                    private static final String ERROR_CONNECTION_RESET = "Connection reset";
293    
294                    /**
295                     * Thread method that waits for input on the client {@link Socket}. Keeps waiting for input while the server is running and no
296                     * PingPong timeout has occurred or the client has gone away.<br />
297                     * <br />
298                     * The input data is read per line, so messages need to end with a newline character '\n'. Since clients need to conform to the
299                     * common denominator, they are required to append a zero (0) byte '\0' as per Flash clients. This character is appended to the
300                     * newline '\n' character and needs to be discarded manually using a {@link BufferedReader#read()}.
301                     * 
302                     * @see SwiftSocketServer#isRunning()
303                     * @see ClientHandler#checkPong()
304                     */
305                    @Override
306                    public void run() {
307                            while (server.isRunning() && (!pingPongMode || checkPong()) && !clientContext.isClientSaidByeBye()) {
308                                    try {
309                                            final String line = clientContext.getClientEndpoint().readLine();
310                                            if (server.isRunning()) {
311                                                    if (line != null) {
312                                                            clientContext.getClientEndpoint().read();
313                                                            handleMessage(clientContext, line);
314                                                    } else {
315                                                            // client closed its connection the hard way
316                                                            LOGGER.debug(String.format("client %s closed connection non-gracefully", clientContext.getClientInetAddress()));
317                                                            server.disposeOfClient(clientContext);
318                                                            break;
319                                                    }
320                                            }
321                                    } catch (final SocketException e) {
322                                            if (e.getMessage().equals(ERROR_SOCKET_CLOSED) || e.getMessage().equals(ERROR_RECV_FAILED)
323                                                            || e.getMessage().equals(ERROR_CONNECTION_RESET)) {
324                                                    if (e.getMessage().equals(ERROR_SOCKET_CLOSED)) {
325                                                            final String msg = "tried to read or write to closed socket (was client %s dropped due to PingPong time-out?)";
326                                                            LOGGER.error(String.format(msg, clientContext.getClientInetAddress()));
327                                                    } else {
328                                                            final String msg = "socket in error state [%s] for client %s";
329                                                            LOGGER.error(String.format(msg, e.getMessage(), clientContext.getClientInetAddress()));
330                                                    }
331                                                    // In case of a socket closed, the client was most likely dropped by the server due to PingPong time-out, which
332                                                    // means bye bye already is set. However, in case the client was dropped due to another reason, manually update the
333                                                    // client to avoid further escalation
334                                                    server.disposeOfClient(clientContext);
335                                                    break;
336                                            }
337                                            // we have no clue what exception was thrown, just log the error and retry
338                                            LOGGER.error(e.getMessage(), e);
339                                    } catch (final IOException e) {
340                                            // we have no clue what exception was thrown, just log the error and retry
341                                            LOGGER.error(e.getMessage(), e);
342                                    }
343    
344                                    ServerUtil.defaultSleep();
345                            }
346                    }
347    
348                    /**
349                     * Handles a single client message string. Creates a {@link ClientMessageToServer} and adds this to the server message queue unless
350                     * it is a server level message, such as a 'pong' notification, in which case the message is executed directly.
351                     * 
352                     * @param clientContext The client that sent the message being interpreted.
353                     * @param datagramMessage The client message string, minus newline character '\n' or zero (0) byte '\0'.
354                     * @see #createDecodedExecutableMessage(ClientContext, String)
355                     * @see SwiftSocketServer#addClientMessage(ClientMessageToServer)
356                     */
357                    public void handleMessage(final ClientContext clientContext, final String datagramMessage) {
358                            final ClientMessageToServer<?> message = createDecodedExecutableMessage(clientContext, datagramMessage);
359                            determineLogger(message).debug(
360                                            String.format("receiving message from %s: %s", clientContext, message.getClass().getSimpleName()));
361                            if (message instanceof ClientMessageToServerPingPong) {
362                                    ((ClientMessageToServerPingPong) message).execute(ClientHandler.this);
363                            } else if (message instanceof ClientMessageToServerByeBye) {
364                                    ((ClientMessageToServerByeBye) message).execute(ClientHandler.this);
365                            } else {
366                                    server.addClientMessage(message);
367                            }
368                    }
369    
370                    /**
371                     * Parses the raw client message into a message id and message content. Passes both values to
372                     * {@link #createDecodedExecutableMessage(ClientContext, int, String)} and returns the result. The first three bytes of the raw
373                     * message denote the message code.
374                     * <p>
375                     * In case anything goes wrong, a {@link ClientMessageToServerInvalid} object is returned instead which generates some useful
376                     * message for logging purposes. The server user can choose to filter out unknown messages at its own discretion. If a
377                     * <code>ClientMessageToServerInvalid</code> is executed however, it will throw a runtime exception with the original message.
378                     * 
379                     * @param clientMessage The client input string that represents a single message.
380                     * @return {@link #createDecodedExecutableMessage(ClientContext, int, String)}
381                     * @see #createInvalidMessage(String, Exception, String)
382                     */
383                    protected ClientMessageToServer<?> createDecodedExecutableMessage(final ClientContext clientContext, final String clientMessage) {
384                            final int messageId;
385                            try {
386                                    messageId = Integer.parseInt(clientMessage.substring(0, MESSAGEID_LENGTH));
387                            } catch (final NumberFormatException e) {
388                                    return createInvalidMessage(clientMessage, e, "invalid message identifier");
389                            }
390                            final String messageContent = clientMessage.substring(MESSAGEID_LENGTH);
391                            return createDecodedExecutableMessage(clientContext, messageId, messageContent);
392                    }
393    
394                    /**
395                     * Creates a {@link ClientMessageToServer} instance based on the message id and message content.
396                     * 
397                     * @param clientContext The client that sent the message, useful to be able to reply to.
398                     * @param messageId The identifier for the message type we need to instantiate.
399                     * @param messageContent The message content for the solved message type to decode.
400                     * @return A {@link ClientMessageToServer} with identified by <code>messageId</code> decoded from <code>messageContent</code>.
401                     * @see SwiftSocketServer#getClientMessageToServerType(int)
402                     * @see ClientMessageToServer#decode(String)
403                     */
404                    private final ClientMessageToServer<?> createDecodedExecutableMessage(final ClientContext clientContext, final int messageId,
405                                    final String messageContent) {
406                            final Class<? extends ClientMessageToServer<?>> messageType = server.getClientMessageToServerType(messageId);
407                            ClientMessageToServer<?> message;
408                            try {
409                                    message = messageType.getConstructor(ClientContext.class).newInstance(clientContext);
410                            } catch (final NoSuchMethodException e) {
411                                    final String errorMessage = "message type '%s' has no valid constructor";
412                                    return createInvalidMessage(messageContent, e, String.format(errorMessage, messageType.getSimpleName()));
413                            } catch (final IllegalAccessException e) {
414                                    return createInvalidMessage(messageContent, e, "constructor not visible for message");
415                            } catch (final InvocationTargetException e) {
416                                    return createInvalidMessage(messageContent, e, "error calling message constructor");
417                            } catch (final InstantiationException e) {
418                                    return createInvalidMessage(messageContent, e, "unable to construct message");
419                            }
420                            message.decode(messageContent);
421                            return message;
422                    }
423    
424                    /**
425                     * Generates a {@link ClientMessageToServerInvalid} object with the failure cause. When executed, this message throws a runtime
426                     * exception with the original exception.
427                     * 
428                     * @param originalMessage The original message
429                     * @param failureCause The exception that occurred when the original messages was being converted into a
430                     *            {@link ClientMessageToServer}.
431                     * @param cause A fine grained exception description when the exception <code>failureCause</code> occurred.
432                     * @return An instance of {@link ClientMessageToServerInvalid}.
433                     */
434                    private ClientMessageToServer<?> createInvalidMessage(final String originalMessage, final Exception failureCause, final String cause) {
435                            final ClientMessageToServerInvalid message = new ClientMessageToServerInvalid(clientContext, failureCause);
436                            message.decode(String.format("%s (original message: '%s')", cause, originalMessage));
437                            return message;
438                    }
439            }
