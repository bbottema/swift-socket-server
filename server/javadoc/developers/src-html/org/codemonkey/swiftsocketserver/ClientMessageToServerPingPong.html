001    package org.codemonkey.swiftsocketserver;
002    
003    /**
004     * Message that decodes pong datagrams. This message is on the server level and therefore only interacts with the {@link ClientHandler} and
005     * is filtered from the client messages queue in the server.
006     * 
007     * @author Benny Bottema
008     * @see ClientMessageToServer
009     * @since 1.0
010     */
011    final class ClientMessageToServerPingPong extends ClientMessageToServer<ClientHandler> {
012    
013            public ClientMessageToServerPingPong(final ClientContext clientContext) {
014                    super(clientContext);
015            }
016    
017            /**
018             * Only validates on zero length, since there is no extra information required for a pong notification.
019             * 
020             * @param messageContent The message string or binary packet that should be empty for <code>ClientMessageToServerPingPong</code>.
021             * @see ClientMessageToServer#decode(String)
022             */
023            @Override
024            protected void decode(final String messageContent) {
025                    if (messageContent.length() > 0) {
026                            throw new RuntimeException("invalid pong message, reason: message too long");
027                    }
028            }
029    
030            /**
031             * Simply delegates the pong event to the {@link ClientHandler}.
032             * 
033             * @param handler The {@link ClientHandler} instance to which the pong notification can be delegated to.
034             * @see ClientHandler#pong()
035             * @see org.codemonkey.swiftsocketserver.ClientMessageToServer#execute(java.lang.Object)
036             */
037            @Override
038            public void execute(final ClientHandler handler) {
039                    handler.pong();
040            }
041    }
