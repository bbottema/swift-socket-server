001    package org.codemonkey.util;
002    
003    /**
004     * Interface for marking objects as datagram decoders. An encoded strings is a string that contains a number that denotes the length of a
005     * value followed by the value itself. Lists are prepended with the number of items in that list followed by a separator.
006     * 
007     * @author Benny Bottema
008     * @since 1.0
009     */
010    public interface DatagramDecoder {
011    
012            /**
013             * Populates the current implementation instance fields with values decoded from the given string (binary packet / datagram) of the
014             * current server response.
015             * 
016             * @param encodedString The string that can be decoded into a concrete class instance.
017             */
018            void decode(StringBuilder encodedString);
019    }
