001    package org.codemonkey.swiftsocketserver;
002    
003    import org.codemonkey.util.DatagramEncoder;
004    
005    /**
006     * Provides a default {@link DatagramEncoder} implementation that returns an empty string. The {@link #encode()} method implemented can be
007     * used (by not overriding it in sub classes) when only the message id is important (for example with Ping Pong messages, which have no
008     * content).
009     * <p>
010     * This class also contains a reference to the current {@link ClientContext} which will receive this message.
011     * <p>
012     * With this base class a message may encode itself to a <code>String</code>. This interface is used to translate messages into so called <a
013     * href="http://gpwiki.org/index.php/Binary_Packet">Binary Packets</a>, or datagrams.
014     * <p>
015     * A response string needs to built up in the following format:<br />
016     * message id (3), encoded string
017     * <p>
018     * The following example could of a login response: <i>0052OK</i><br />
019     * The following example could of a pong notification: <i>{@value SwiftSocketServer#MESSAGE_ID_PINGPONG}</i><br />
020     * 
021     * @author Benny Bottema
022     * @see SwiftSocketServer#addClientMessage(ClientMessageToServer)
023     * @see SwiftSocketServer#getNextClientMessage()
024     * @see DatagramEncoder#encode()
025     * @since 1.0
026     */
027    public abstract class ServerMessageToClient implements DatagramEncoder {
028    
029            /**
030             * Holds the information about the whereabouts of the connected client that was responsible for the current message.
031             */
032            private final ClientContext clientContext;
033    
034            /**
035             * Constructor; simply stores a reference to the client associated with this message.
036             * 
037             * @param clientContext The client associated with this message.
038             */
039            protected ServerMessageToClient(final ClientContext clientContext) {
040                    this.clientContext = clientContext;
041            }
042    
043            /**
044             * @return {@link #clientContext}
045             */
046            public ClientContext getClientContext() {
047                    return clientContext;
048            }
049    
050            /**
051             * Default implementation which returns an empty string. This is useful for messages that have no parameters, just a message code.
052             * 
053             * @return An empty string.
054             * @see DatagramEncoder#encode()
055             */
056            @Override
057            public String encode() {
058                    return "";
059            }
060    }
