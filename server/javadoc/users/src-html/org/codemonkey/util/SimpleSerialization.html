001    package org.codemonkey.util;
002    
003    import java.io.Serializable;
004    
005    /**
006     * Guarantees an object can produce and populate itself with a string value, not unlike deserialization.
007     * <p>
008     * This class is like {@link Serializable} except it's a much simpler version where the object itself provides the serialization code.
009     * 
010     * @author Benny Bottema
011     * @param <T> The type of object produced by {@link #deserialize(String)}.
012     * @see #deserialize(String)
013     * @since 1.0
014     */
015    public interface SimpleSerialization<T> {
016            /**
017             * Makes the object convert itself to a string representation.
018             * 
019             * @return A <code>String</code> representation of the current <code>Object</code> that can be used to recreate an instance of the same
020             *         type using {@link #deserialize(String)}.
021             */
022            String serialize();
023    
024            /**
025             * The general contract of <code>deserialize</code> is:<br />
026             * Instance <code>T a</code> equals <code>T b</code> where <code>b = new T().deserialize(a.serialize())</code>.
027             * 
028             * @param value A String value as produced by {@link #serialize()}.
029             * @return A populated Object from a String produced by {@link #serialize()}.
030             */
031            T deserialize(String value);
032    }
